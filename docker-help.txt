cat > Dockerfile << 'EOF'
> FROM node:20-slim
> 
> WORKDIR /app
> 
> # Устанавливаем корневые зависимости (google-auth-library)
> COPY package*.json ./
> RUN npm install
> 
> # Устанавливаем зависимости сервера
> COPY server/package*.json ./server/
> RUN cd server && npm install
> 
> # Копируем остальной код
> COPY server ./server
> COPY client ./client
> 
> WORKDIR /app/server
> 
> EXPOSE 5000
> 
> # Используем production-скрипт
> CMD ["npm", "start"]
> EOF
